Clone with Properties

RFC
PHP < 8.5 
final readonly class PhpVersion {
    public function __construct(
        public string $version = 'PHP 8.4',
    ) {}

    public function withVersion(string $version) {
        $newObject = clone $this;
        $newObject->version = $version;
        return $newObject;
    }
}

// Fatal error: Cannot modify readonly property
PHP 8.5 
final readonly class PhpVersion {
    public function __construct(
        public string $version = 'PHP 8.4',
    ) {}

    public function withVersion(string $version) {
        return clone($this, ['version' => $version]);
    }
}

$version = new PhpVersion();
var_dump($version->withVersion('PHP 8.5'));

Pipe Operator

RFC
PHP < 8.5 
$input = ' Some kind of string. ';

$output = strtolower(
    str_replace(['.', '/', '‚Ä¶'], '',
        str_replace(' ', '-',
            trim($input)
        )
    )
);

// Nested, hard to read
PHP 8.5 
$input = ' Some kind of string. ';

$output = $input
    |> trim(...)
    |> (fn(string $s) => str_replace(' ', '-', $s))
    |> (fn(string $s) => str_replace(['.', '/', '‚Ä¶'], '', $s))
    |> strtolower(...);

// Clean, readable flow!

First Class Callables in Constant Expressions

RFC RFC 2
PHP < 8.5 
final class CalculatorTest
{
    #[DataProvider('subtractionProvider')]
    public function testSubtraction(
        int $minuend,
        int $subtrahend,
        int $result
    ): void {
        $this->assertSame($result,
            Calculator::subtract($minuend, $subtrahend));
    }

    public static function subtractionProvider(): iterable {
        // Separate method required
    }
}
PHP 8.5 
final class CalculatorTest
{
    #[Test\CaseGenerator(static function(): iterable {
        for ($i = -10; $i <= 10; $i++) {
            yield [$i, $i, 0];
            yield [$i, 0, $i];
            yield [0, $i, ($i * -1)];
        }
    })]
    public function testSubtraction(int $minuend, int $subtrahend, int $result) {
        assert(Calculator::subtract($minuend, $subtrahend) === $result);
    }
}

Persistent cURL Share Handle

RFC RFC 2
PHP < 8.5 
$sh = curl_share_init();
curl_share_setopt($sh, CURLSHOPT_SHARE, CURL_LOCK_DATA_DNS);
curl_share_setopt($sh, CURLSHOPT_SHARE, CURL_LOCK_DATA_CONNECT);

$ch1 = curl_init('https://php.net/');
curl_setopt($ch1, CURLOPT_SHARE, $sh);

$ch2 = curl_init('https://thephp.foundation/');

PHP 8.5 
$sh = curl_share_init_persistent([
    CURL_LOCK_DATA_DNS,

$ch1 = curl_init('https://php.net/');

curl_exec($ch1);

curl_setopt($ch2, CURLOPT_SHARE, $sh);
curl_exec($ch2);

curl_close($ch1);
curl_close($ch2);

New array_first() and array_last() Functions

RFC
PHP < 8.5 
$php = [
    'php-83' => ['state' => 'active', 'branch' => 'PHP-8.3'],
    'php-84' => ['state' => 'active', 'branch' => 'PHP-8.4'],
    'php-85' => ['state' => 'upcoming', 'branch' => 'PHP-8.5'],
];

$upcomingRelease = null;
foreach ($php as $key => $version) {
    if ($version['state'] === 'upcoming') {
        $upcomingRelease = $version;
        break;
    }
}

var_dump($upcomingRelease);
PHP 8.5 
$php = [
    'php-83' => ['state' => 'active', 'branch' => 'PHP-8.3'],
    'php-84' => ['state' => 'active', 'branch' => 'PHP-8.4'],
    'php-85' => ['state' => 'upcoming', 'branch' => 'PHP-8.5'],
];

$upcomingRelease = array_first(
    array_filter(
        $php,
        static fn($v) => $v['state'] === 'upcoming'
    )
);

var_dump($upcomingRelease);
